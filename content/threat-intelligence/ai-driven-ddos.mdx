{`import numpy as np

def detect_anomaly(latency_array):
    # Calculate dynamic threshold based on standard deviation
    threshold = np.mean(latency_array) + (3 * np.std(latency_array))
    anomalies = []
    
    for val in latency_array:
        if val > threshold:
            anomalies.append(val)
            
    return anomalies`}
